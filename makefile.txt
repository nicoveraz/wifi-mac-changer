# Makefile for Wi-Fi MAC Changer

# Variables
SCRIPT_NAME = wifi-mac-changer
INSTALL_DIR = /usr/local/bin

.PHONY: install uninstall

install:
	@echo "Installing Wi-Fi MAC Changer..."
	@cp $(SCRIPT_NAME) $(INSTALL_DIR)/
	@chmod +x $(INSTALL_DIR)/$(SCRIPT_NAME)
	@echo "Installation complete. You can now use 'wifi-mac-changer' command."

uninstall:
	@echo "Uninstalling Wi-Fi MAC Changer..."
	@rm -f $(INSTALL_DIR)/$(SCRIPT_NAME)
	@echo "Uninstallation complete."
